Besoin d'aide?

Voici la liste des différentes opérations possibles;

? op = + (ou un espace) & x = (nombre entier) & y = (nombre entier)
return {"op":"+", "x":"nombre entier", "y":"nombre entier", "value":"x + y"}

? op = - & x = (nombre entier) & y = (nombre entier)
return {"op":"-", "x":"nombre entier", "y":"nombre entier", "value":"x - y"}

? op = * & x = (nombre entier) & y = (nombre entier)
return {"op":"*", "x":"nombre entier", "y":"nombre entier", "value":"x * y"}

? op = / & x = (nombre entier) & y = (nombre entier)
return {"op":"/", "x":"nombre entier", "y":"nombre entier", "value":"x / y"}

? op = % & x = (nombre entier) & y = (nombre entier)
return {"op":"%", "x":"nombre entier", "y":"nombre entier", "value":"x % y"}

? op = ! & n = (nombre entier)
return {"op":"!", "n":"nombre entier", "value":"n!"}

? op = p & n = (nombre entier)
return {"op":"p", "n":"nombre entier", "value":"true si n est un nombre premier"}

? op = np & n = (nombre entier)
return {"op":"n", "n":"nombre entier", "value":"Le nombre premier le plus proche"}
